057/**
058 * Class to test HBaseHbck.
059 * Spins up the minicluster once at test start and then takes it down afterward.
060 * Add any testing of HBaseHbck functionality here.
061 */
062@Category({LargeTests.class, ClientTests.class})
063public class TestHbck {
064  @ClassRule
065  public static final HBaseClassTestRule CLASS_RULE =
066      HBaseClassTestRule.forClass(TestHbck.class);
067
068  private static final Logger LOG = LoggerFactory.getLogger(TestHbck.class);
069  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
070
071  @Rule
072  public TestName name = new TestName();
073
074  private static final TableName TABLE_NAME = TableName.valueOf(TestHbck.class.getSimpleName());
075
076  private static ProcedureExecutor<MasterProcedureEnv> procExec;
077
078  @BeforeClass
079  public static void setUpBeforeClass() throws Exception {
080    TEST_UTIL.startMiniCluster(3);
081    TEST_UTIL.createMultiRegionTable(TABLE_NAME, Bytes.toBytes("family1"), 5);
082    procExec = TEST_UTIL.getMiniHBaseCluster().getMaster().getMasterProcedureExecutor();
083  }
084
085  @AfterClass
086  public static void tearDownAfterClass() throws Exception {
087    TEST_UTIL.shutdownMiniCluster();
088  }
089
090  public static class SuspendProcedure extends
091      ProcedureTestingUtility.NoopProcedure<MasterProcedureEnv> implements TableProcedureInterface {
092    public SuspendProcedure() {
093      super();
094    }
095
096    @Override
097    protected Procedure[] execute(final MasterProcedureEnv env)
098        throws ProcedureSuspendedException {
099      // Always suspend the procedure
100      throw new ProcedureSuspendedException();
101    }
102
103    @Override
104    public TableName getTableName() {
105      return TABLE_NAME;
106    }
107
108    @Override
109    public TableOperationType getTableOperationType() {
110      return TableOperationType.READ;
111    }
112  }
113
114  @Test
115  public void testBypassProcedure() throws Exception {
116    // SuspendProcedure
117    final SuspendProcedure proc = new SuspendProcedure();
118    long procId = procExec.submitProcedure(proc);
119    Thread.sleep(500);
120
121    //bypass the procedure
122    List<Long> pids = Arrays.<Long>asList(procId);
123    List<Boolean> results =
124        TEST_UTIL.getHbck().bypassProcedure(pids, 30000, false, false);
125    assertTrue("Failed to by pass procedure!", results.get(0));
126    TEST_UTIL.waitFor(5000, () -> proc.isSuccess() && proc.isBypass());
127    LOG.info("{} finished", proc);
128  }

130  @Test
131  public void testSetTableStateInMeta() throws IOException {
132    Hbck hbck = TEST_UTIL.getHbck();
133    // set table state to DISABLED
134    hbck.setTableStateInMeta(new TableState(TABLE_NAME, TableState.State.DISABLED));
135    // Method {@link Hbck#setTableStateInMeta()} returns previous state, which in this case
136    // will be DISABLED
137    TableState prevState =
138        hbck.setTableStateInMeta(new TableState(TABLE_NAME, TableState.State.ENABLED));
139    assertTrue("Incorrect previous state! expeced=DISABLED, found=" + prevState.getState(),
140        prevState.isDisabled());
141  }

143  @Test
144  public void testAssigns() throws IOException {
145    Hbck hbck = TEST_UTIL.getHbck();
146    try (Admin admin = TEST_UTIL.getConnection().getAdmin()) {
147      List<RegionInfo> regions = admin.getRegions(TABLE_NAME);
148      for (RegionInfo ri: regions) {
149        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager().
150            getRegionStates().getRegionState(ri.getEncodedName());
151        LOG.info("RS: {}", rs.toString());
152      }
153      List<Long> pids = hbck.unassigns(regions.stream().map(r -> r.getEncodedName()).
154          collect(Collectors.toList()));
155      waitOnPids(pids);
156      for (RegionInfo ri: regions) {
157        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager().
158            getRegionStates().getRegionState(ri.getEncodedName());
159        LOG.info("RS: {}", rs.toString());
160        assertTrue(rs.toString(), rs.isClosed());
161      }
162      pids = hbck.assigns(regions.stream().map(r -> r.getEncodedName()).
163          collect(Collectors.toList()));
164      waitOnPids(pids);
165      for (RegionInfo ri: regions) {
166        RegionState rs = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager().
167            getRegionStates().getRegionState(ri.getEncodedName());
168        LOG.info("RS: {}", rs.toString());
169        assertTrue(rs.toString(), rs.isOpened());
170      }
171      // What happens if crappy region list passed?
172      pids = hbck.assigns(Arrays.stream(new String [] {"a", "some rubbish name"}).
173          collect(Collectors.toList()));
174      for (long pid: pids) {
175        assertEquals(org.apache.hadoop.hbase.procedure2.Procedure.NO_PROC_ID, pid);
176      }
177    }
178  }

180  @Test
181  public void testScheduleSCP() throws Exception {
182    HRegionServer testRs = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME);
183    TEST_UTIL.loadTable(TEST_UTIL.getConnection().getTable(TABLE_NAME), Bytes.toBytes("family1"),
184      true);
185    ServerName serverName = testRs.getServerName();
186    Hbck hbck = TEST_UTIL.getHbck();
187    List<Long> pids =
188        hbck.scheduleServerCrashProcedure(Arrays.asList(ProtobufUtil.toServerName(serverName)));
189    assertTrue(pids.get(0) > 0);
190    LOG.info("pid is {}", pids.get(0));
191
192    pids = hbck.scheduleServerCrashProcedure(Arrays.asList(ProtobufUtil.toServerName(serverName)));
193    assertTrue(pids.get(0) == -1);
194    LOG.info("pid is {}", pids.get(0));
195  }
196
197  private void waitOnPids(List<Long> pids) {
198    for (Long pid: pids) {
199      while (!TEST_UTIL.getHBaseCluster().getMaster().getMasterProcedureExecutor().
200          isFinished(pid)) {
201        Threads.sleep(100);
202      }
203    }
204  }
